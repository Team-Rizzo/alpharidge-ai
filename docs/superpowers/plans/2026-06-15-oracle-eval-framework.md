# Oracle + Core Eval Framework Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the GLM 5.1 + Opus 4.6 oracle over CC-NEWS and a reusable framework that scores any candidate analyzer on per-field accuracy, cost, and latency.

**Architecture:** A standalone `eval/` Python package at `/home/rizzo/talisman/eval` that imports `alpharidge_ai` for the `ArticleIntelligence` schema and `scoring.py` similarity helpers. Candidate analyzers implement a common `Analyzer` protocol; gold data implements a `GoldSet` protocol; a `FieldMetric` registry scores predictions field-by-field; a `Runner` produces an `EvalReport`. The oracle is generated by running the existing two-call analyzer with the oracle model (GLM/Opus) over a finance-dense English CC-NEWS sample.

**Tech Stack:** Python 3.13, pytest, the existing `alpharidge_ai` analyzer/scoring, `warcio`, `langdetect`, `sentence-transformers` (already installed), `rapidfuzz` (Levenshtein), `rouge-score`, OpenRouter (GLM) + Anthropic-compatible (Opus) via the analyzer's OpenAI client.

**Scope:** Core framework + CC-NEWS oracle + GLM-vs-Opus calibration. **Out of scope (Plan 2):** the 8 external HF dataset adapters and external calibration. **Out of scope (sub-projects #2/#3):** candidate ML models, validator/API/miner changes.

from eval.metrics.base import MetricResult

LADDER = ["very_bullish", "bullish", "slightly_bullish", "neutral",
          "slightly_bearish", "bearish", "very_bearish"]
_IDX = {v: i for i, v in enumerate(LADDER)}
_MAX = len(LADDER) - 1

class OrdinalSentimentMetric:
    def score(self, pred, gold) -> MetricResult:
        if pred not in _IDX or gold not in _IDX:
            return MetricResult(0.0)
        dist = abs(_IDX[pred] - _IDX[gold])
        return MetricResult(1.0 - dist / _MAX, {"distance": dist})

from eval.metrics.base import MetricResult

class NumericMetric:
    def __init__(self, tolerance: float = 0.1):
        self.tol = tolerance
    def score(self, pred, gold) -> MetricResult:
        if pred is None or gold is None:
            return MetricResult(0.0)
        err = abs(float(pred) - float(gold))
        if err <= self.tol:
            return MetricResult(1.0)
        if err >= 2 * self.tol:
            return MetricResult(0.0)
        return MetricResult(1.0 - (err - self.tol) / self.tol, {"abs_err": err})

# eval/metrics/liststruct.py
from statistics import mean
from eval.metrics.base import MetricResult

class ListStructMetric:
    def __init__(self, key, compare_fields=None):
        self.key = key
        self.compare_fields = compare_fields or []
    def score(self, pred, gold) -> MetricResult:
        pred = pred or []; gold = gold or []
        pk = {self.key(d): d for d in pred}
        gk = {self.key(d): d for d in gold}
        ps, gs = set(pk), set(gk)
        tp = len(ps & gs); fp = len(ps - gs); fn = len(gs - ps)
        if tp == fp == fn == 0:
            return MetricResult(1.0)
        f1 = (2 * tp) / (2 * tp + fp + fn) if (2 * tp + fp + fn) else 0.0
        if not self.compare_fields or not (ps & gs):
            return MetricResult(f1, {"f1": f1})
        agrees = []
        for k in (ps & gs):
            agrees.append(mean(1.0 if pk[k].get(f) == gk[k].get(f) else 0.0
                               for f in self.compare_fields))
        field_agree = mean(agrees)
        return MetricResult(mean([f1, field_agree]), {"f1": f1, "field_agreement": field_agree})
```

# eval/metrics/text.py
from rouge_score import rouge_scorer
from eval.metrics.base import MetricResult

_ROUGE = rouge_scorer.RougeScorer(["rougeL"], use_stemmer=True)
_EMBED = None

def _embedder():
    global _EMBED
    if _EMBED is None:
        from sentence_transformers import SentenceTransformer
        _EMBED = SentenceTransformer("all-MiniLM-L6-v2")
    return _EMBED

class TextMetric:
    def __init__(self, use_embedding: bool = True, w_rouge: float = 0.5):
        self.use_embedding = use_embedding
        self.w_rouge = w_rouge
    def score(self, pred, gold) -> MetricResult:
        if not pred or not gold:
            return MetricResult(0.0)
        rouge = _ROUGE.score(gold, pred)["rougeL"].fmeasure
        if not self.use_embedding:
            return MetricResult(rouge, {"rougeL": rouge})
        import numpy as np
        e = _embedder().encode([pred, gold], normalize_embeddings=True)
        cos = float(np.dot(e[0], e[1]))
        val = self.w_rouge * rouge + (1 - self.w_rouge) * cos
        return MetricResult(val, {"rougeL": rouge, "cosine": cos})
```

# eval/metrics/embedding.py
import numpy as np
from eval.metrics.base import MetricResult
from alpharidge_ai.analyzer.scoring import _jaccard

class EmbeddingCosineMetric:
    def score(self, pred, gold) -> MetricResult:
        if not pred or not gold:
            return MetricResult(0.0)
        a, b = np.asarray(pred, float), np.asarray(gold, float)
        na, nb = np.linalg.norm(a), np.linalg.norm(b)
        if na == 0 or nb == 0:
            return MetricResult(0.0)
        return MetricResult(max(0.0, float(np.dot(a, b) / (na * nb))))

class KeywordJaccardMetric:
    def score(self, pred, gold) -> MetricResult:
        ps = {str(x).lower() for x in (pred or [])}
        gs = {str(x).lower() for x in (gold or [])}
        return MetricResult(float(_jaccard(ps, gs)))
```

        self._items = None

# eval/runner.py
from dataclasses import dataclass, field
from statistics import mean
from eval.metrics.fields import FIELD_METRICS, score_field

@dataclass
class EvalReport:
    model: str
    n: int
    field_accuracy: dict          # field -> mean score
    overall: float
    avg_input_tokens: float
    avg_output_tokens: float
    latency_p50: float
    latency_p95: float
    per_item: list = field(default_factory=list)

def _pct(values, p):
    if not values:
        return 0.0
    s = sorted(values)
    k = max(0, min(len(s) - 1, int(round((p / 100) * (len(s) - 1)))))
    return s[k]

class Runner:
    def run(self, analyzer, goldset) -> EvalReport:
        scored = {f: [] for f in goldset.fields() if f in FIELD_METRICS}
        in_tok, out_tok, lat = [], [], []
        n = 0
        for item in goldset:
            res = analyzer.analyze(item.article)
            in_tok.append(res.input_tokens); out_tok.append(res.output_tokens)
            lat.append(res.latency_ms); n += 1
            for f in scored:
                if f in item.labels:
                    scored[f].append(score_field(f, res.intel, item.labels).value)
        field_acc = {f: (mean(v) if v else 0.0) for f, v in scored.items()}
        overall = mean(field_acc.values()) if field_acc else 0.0
        return EvalReport(
            model=getattr(analyzer, "model", "?"), n=n, field_accuracy=field_acc,
            overall=overall,
            avg_input_tokens=mean(in_tok) if in_tok else 0.0,
            avg_output_tokens=mean(out_tok) if out_tok else 0.0,
            latency_p50=_pct(lat, 50), latency_p95=_pct(lat, 95),
        )

import json
from dataclasses import asdict

PRICES = {  # $ per MTok (input, output)
    "glm-5.1": (0.98, 3.08),
    "claude-opus-4-6": (5.0, 25.0),
    "claude-haiku-4-5": (1.0, 5.0),
}

def cost_per_article(rep, model_key=None) -> float:
    pin, pout = PRICES.get(model_key or rep.model, (0.0, 0.0))
    return rep.avg_input_tokens / 1e6 * pin + rep.avg_output_tokens / 1e6 * pout

def render_markdown(reports) -> str:
    fields = sorted({f for r in reports for f in r.field_accuracy})
    head = ["model", "overall", "cost/article", "p50ms", "p95ms"] + fields
    lines = ["| " + " | ".join(head) + " |", "|" + "---|" * len(head)]

               f"{r.latency_p50:.0f}", f"{r.latency_p95:.0f}"]
        row += [f"{r.field_accuracy.get(f, float('nan')):.3f}" for f in fields]
        lines.append("| " + " | ".join(row) + " |")
    return "\n".join(lines)

def to_json(reports) -> str:
    return json.dumps([{**asdict(r), "cost_per_article": cost_per_article(r)} for r in reports],

import json, random, sys
from pathlib import Path
sys.path.insert(0, "/home/rizzo/talisman/news-scraper")
import ccnews_ingest as cc
from langdetect import detect, DetectorFactory
DetectorFactory.seed = 0

ASSETS_JSON = "/home/rizzo/talisman/alpharidge-ai/alpharidge_ai/analyzer/data/assets.json"

def load_gazetteer(path=ASSETS_JSON) -> set:
    data = json.loads(Path(path).read_text())
    terms = set()
    items = data.values() if isinstance(data, dict) else data
    for a in items:
        for key in ("symbol", "ticker", "name", "cashtag"):
            v = a.get(key) if isinstance(a, dict) else None
            if v:
                terms.add(str(v).lower())
    terms |= {"fed", "fomc", "cpi", "earnings", "sec", "etf", "rate", "inflation"}
    return {t for t in terms if len(t) >= 2}

def is_english(text: str) -> bool:
    try:
        return detect(text[:1500]) == "en"
    except Exception:
        return False

def finance_hits(text: str, gaz: set) -> int:
    toks = {w.strip(".,!?:;()[]$").lower() for w in text.split()}
    return len(toks & gaz)

def should_keep(text, gaz, rand_slice: bool, min_hits: int = 1) -> bool:
    return finance_hits(text, gaz) >= min_hits or rand_slice

def sample(warc_paths, n=5000, rand_fraction=0.2, seed=0):
    """Yield up to n Article-dicts from local WARCs."""
    from warcio.archiveiterator import ArchiveIterator
    rng = random.Random(seed)
    gaz = load_gazetteer()
    out, aid = [], 0
    for wp in warc_paths:
        with open(wp, "rb") as fh:
            for rec in ArchiveIterator(fh):
                if rec.rec_type != "response":
                    continue
                cb = rec.content_stream().read(cc.MAX_CONTENT_READ)
                if len(cb) < 200:
                    continue
                title, text = cc._extract_text_from_html(cb.decode("utf-8", "replace"))
                if not text or len(text) < cc.MIN_TEXT_LEN or not is_english(f"{title}. {text}"):
                    continue
                if not should_keep(f"{title} {text}", gaz, rng.random() < rand_fraction):
                    continue
                url = str(rec.rec_headers.get_header("WARC-Target-URI") or "")
                aid += 1
                out.append({"id": aid, "url": url[:2000], "title": title[:500],
                            "source": "ccnews", "published": cc._extract_date_from_warc(rec) or None,
                            "summary": None, "content": (f"{title}. {text}")[:cc.MAX_TEXT_LEN]})
                if len(out) >= n:
                    return out
    return out

import json
from statistics import mean
from pathlib import Path
from eval.metrics.fields import FIELD_METRICS, score_field

def _load(path):
    out = {}
    for line in Path(path).read_text().splitlines():
        if line.strip():
            d = json.loads(line)
            out[d["article"]["id"]] = d["labels"]
    return out

def agreement(glm_path, opus_path) -> dict:
    glm, opus = _load(glm_path), _load(opus_path)
    ids = sorted(set(glm) & set(opus))
    per_field = {}
    for f in FIELD_METRICS:
        vals = [score_field(f, glm[i], opus[i]).value
                for i in ids if f in glm[i] and f in opus[i]]
        if vals:
            per_field[f] = round(mean(vals), 4)
    return {"n_overlap": len(ids), "fields": per_field,
            "overall": round(mean(per_field.values()), 4) if per_field else 0.0}

# eval/cli.py
import argparse, json
from pathlib import Path

def build_parser():
    p = argparse.ArgumentParser("eval")
    sub = p.add_subparsers(dest="cmd", required=True)
    s = sub.add_parser("sample"); s.add_argument("--warcs", nargs="+", required=True)
    s.add_argument("--n", type=int, default=5000); s.add_argument("--out", required=True)
    l = sub.add_parser("label"); l.add_argument("--sample", required=True)
    l.add_argument("--model", required=True); l.add_argument("--out-dir", required=True)
    l.add_argument("--llm-base"); l.add_argument("--limit", type=int)
    c = sub.add_parser("calibrate"); c.add_argument("--glm", required=True)
    c.add_argument("--opus", required=True); c.add_argument("--out", required=True)
    r = sub.add_parser("run"); r.add_argument("--analyzer", required=True)
    r.add_argument("--gold", required=True); r.add_argument("--out")
    return p

def main(argv=None):
    ns = build_parser().parse_args(argv)
    if ns.cmd == "sample":
        from eval.oracle.sampler import sample
        arts = sample(ns.warcs, n=ns.n)
        Path(ns.out).write_text("\n".join(json.dumps(a) for a in arts))
        print(f"sampled {len(arts)} -> {ns.out}")
    elif ns.cmd == "label":
        from eval.oracle.labeler import OracleConfig, build_analyzer, label_articles
        arts = [json.loads(x) for x in Path(ns.sample).read_text().splitlines() if x.strip()]
        if ns.limit:
            arts = arts[:ns.limit]
        cfg = OracleConfig(model=ns.model, out_dir=Path(ns.out_dir), llm_base=ns.llm_base)
        print(label_articles(arts, build_analyzer(cfg), cfg))
    elif ns.cmd == "calibrate":
        from eval.oracle.calibrate import agreement
        Path(ns.out).write_text(json.dumps(agreement(ns.glm, ns.opus), indent=2))
        print(f"calibration -> {ns.out}")
    elif ns.cmd == "run":
        from eval.gold.oracle_goldset import OracleGoldSet
        from eval.runner import Runner
        from eval.report import render_markdown
        # analyzer "oracle:<model>" replays a gold file as a perfect analyzer (baseline)
        from eval.analyzers.base import AnalyzerResult
        gold = OracleGoldSet(ns.gold)
        class Replay:
            model = ns.analyzer
            _m = {gi.article.id: gi.labels for gi in gold}
            def analyze(self, a): return AnalyzerResult(intel=self._m.get(a.id, {}))
        rep = Runner().run(Replay(), gold)
        print(render_markdown([rep]))

if __name__ == "__main__":
    main()
```
